  1. Consider the following Java class: 
        public class Book {
	
        private String title;
        private Integer numPages;
		
        public Book( String t, Integer n ) {
            title = t;
            numPages = n;
        }
		
        public String toString() {
            return "This book's title is " + title + " and it contains " + numPages + " pages";
        }
		
        public String getTitle() {
            return title;
        }
		
        public Integer getNumPages() {
            return numPages;
        }
		
        public void setTitle( String t ) {
            title = t;
        }
		
        public void setNumPages( Integer n ) {
            numPages = n;
        }
		
    } // end of class Book
